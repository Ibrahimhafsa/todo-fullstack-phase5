2. Access the frontend locally using port-forwarding:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "todo-chatbot.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app=frontend" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.targetPort }}

  Then visit: http://localhost:{{ .Values.frontend.service.port }}
